144 {

146 // Texto con las etiquetas html que contiene los registros eliminados
147 $regs = '<div class="row">';

149 foreach ($del as $attr => $value) {

152 str_contains($attr, 'key') ||
153 str_contains($attr, 'token') ||
154 empty($value)
155 ) {
156 continue;
157 }
158 $regs .= "<div class='col-6 break-words'><b>$attr:</b> $value</div>";
159 }
160 $regs .= '</div>';
161 array_push($trashed, [
162 'id' => secure_record($del->id),
163 'deleted_at' => Carbon::parse($del->deleted_at)->format("d-m-Y"),
164 'module' => $model_name ?? get_class($del),
165 'registers' => $regs
166 ]);
167 }
168 return $trashed;
