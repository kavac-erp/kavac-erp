164  //Se busca el compromiso asociado a la emisión de pago

166  $compromise = \Modules\Budget\Models\BudgetCompromise::query()
167  ->find($bugetStage->budget_compromise_id);
168 
169  if (isset($compromise)) {
170  if (Module::has('Payroll') && Module::isEnabled('Payroll')) {
171  $CodePayroll = \App\Models\CodeSetting::where("model", \Modules\Payroll\Models\Payroll::class)->first();
172  if (isset($CodePayroll)) {
173  $pattern = '/' . $CodePayroll->format_prefix . '/';
174  if (preg_match($pattern, $compromise->document_number)) {
175  $payroll = \Modules\Payroll\Models\Payroll::query()
176  ->where(
177  'code',
178  $compromise->document_number
179  )->first();
180 
181  if (isset($payroll)) {
182  $payrollPaymentPeriod = $payroll->payrollPaymentPeriod;

184  return $payrollPaymentPeriod->payrollPaymentType->order;
