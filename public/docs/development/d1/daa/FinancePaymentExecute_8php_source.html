186 ->where('type', 'PAG')->first();
187
188 //Se busca el compromiso asociado a la emisión de pago
189 if (isset($bugetStage)) {
190 $compromise = \Modules\Budget\Models\BudgetCompromise::query()
191 ->find($bugetStage->budget_compromise_id);
192
193 if (isset($compromise)) {
194 if (Module::has('Payroll') && Module::isEnabled('Payroll')) {
195 $CodePayroll = \App\Models\CodeSetting::where("model", \Modules\Payroll\Models\Payroll::class)->first();
196 if (isset($CodePayroll)) {
197 $pattern = '/' . $CodePayroll->format_prefix . '/';
198 if (preg_match($pattern, $compromise->document_number)) {
199 $payroll = \Modules\Payroll\Models\Payroll::query()
200 ->where(
201 'code',
202 $compromise->document_number
203 )->first();
204
205 if (isset($payroll)) {
206 $payrollPaymentPeriod = $payroll->payrollPaymentPeriod;
207 if (isset($payrollPaymentPeriod)) {
208 return $payrollPaymentPeriod->payrollPaymentType->order;

315 /**
316 * Scope para buscar y filtrar datos de emisiones de pago
317 *
318 * @method scopeSearch
319 *
320 * @author Daniel Contreras <dcontreras@cenditel.gob.ve>
321 *
322 * @param \Illuminate\Database\Eloquent\Builder $query Objeto con la consulta
323 * @param string $search Cadena de texto a buscar
324
325 * @return \Illuminate\Database\Eloquent\Builder
326 */
327 public function scopeSearch($query, $search)
328 {
329 return $query
330 ->where(DB::raw('upper(code)'), 'LIKE', '%' . strtoupper($search) . '%')
331 ->orWhereRaw("TO_CHAR(paid_at, 'DD/MM/YYYY') LIKE '%" . strtoupper($search) . "%'")
332 ->orWhere(DB::raw('upper(observations)'), 'LIKE', '%' . strtoupper($search) . '%');
