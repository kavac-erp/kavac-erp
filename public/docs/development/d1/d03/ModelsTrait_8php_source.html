287  foreach ($results as $result) {

291 
292  $filename = $result;
293 
294  if (is_dir($filename)) {

299  }

301  /** Escanea los directorios de módulos para obtener los correspondientes modelos */
302  $results_modules = scandir($modules_path);
303  foreach ($results_modules as $result_module) {
304  if (

306  !file_exists(base_path() . '/modules/' . $result_module . '/Models')
