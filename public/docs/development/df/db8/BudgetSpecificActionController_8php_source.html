394
395 if ($source === 'report') {
396 if ($existsFormulation) {
397 array_push($data, [
398 'id' => $specificAction->id,
399 'text' => $specificAction->code . " - " . $specificAction->name
400 ]);
401 }
402 } else {
403 if (!$existsFormulation) {
404 array_push($data, [
405 'id' => $specificAction->id,
406 'text' => $specificAction->code . " - " . $specificAction->name
407 ]);
408 }
409 }
410 }
411
412 return response()->json($data);
413 }

415 /**
416 * Obtiene todas las acciones específicas agrupadas por Proyectos y Acciones Centralizadas
417 *
418 * @author Ing. Roldan Vargas <rvargas@cenditel.gob.ve> | <roldandvg@gmail.com>
419 *
420 * @param string $formulated_year Año de formulación por el cual filtrar la información
421 * @param boolean $formulated Indica si se debe validar con una formulación de presupuesto
422 * @param integer $institutionId Identificador de la institución
423 * @param string $selDate Fecha en la cual se esta realizando la consulta
424 *
425 * @return \Illuminate\Http\JsonResponse
426 */
427 public function getGroupAllSpecificActions(
428 $formulated_year = '',
429 $formulated = null,
430 $institutionId = null,
431 $selDate = null
432 ) {
433 if ($formulated_year && strlen($formulated_year) > 4) {
434 try {
435 $formulated_year = Crypt::decrypt($formulated_year);
436 } catch (DecryptException $e) {
437 Log::error($e->getMessage());
438 }
439 }
440 /* Arreglo que contiene las acciones específicas agrupadas por proyectos */
441 $dataProjects = ['text' => 'Proyectos', 'children' => []];
442 /* Arreglo que contiene las acciones específicas agrupadas por acciones centralizadas */
443 $dataCentralizedActions = ['text' => 'Acciones Centralizadas', 'children' => []];
444
445 /* Arreglo que contiene las acciones específicas */
446 $data = [['id' => '', 'text' => 'Seleccione...']];
447
448 $budgetSpecificAction = BudgetSpecificAction::has('specificable');
449
450 if (!is_null($selDate)) {
451 $budgetSpecificAction = $budgetSpecificAction->where(
452 'from_date',
453 '>=',
454 $selDate
455 )->where('to_date', '<=', $selDate);
456 }
457
458 /* Objeto que contiene información de las acciones específicas */
459 $budgetSpecificAction = ($formulated_year)
460 ? $budgetSpecificAction->whereYear('from_date', $formulated_year)
461 ->orWhereYear('to_date', $formulated_year)
462 : $budgetSpecificAction;
463
464 /* Objeto con información de las acciones específicas a consultar */
465 $budgetSpecificAction = (is_null($institutionId))
466 ? $budgetSpecificAction
467 : $budgetSpecificAction->whereHas('specificable', function ($q) use ($institutionId) {
468 $q->whereHas('department', function ($qq) use ($institutionId) {
469 $qq->where('institution_id', $institutionId);
470 });
471 });
472
473 $sp_accs = $budgetSpecificAction->get();
474
475 $withoutFormulations = false;
476 $hasFormulations = false;
477
478 /* Agrega las acciones específicas para cada grupo */
479 foreach ($sp_accs as $sp_acc) {
480 $filter = (!is_null($formulated) && $formulated) ? BudgetSubSpecificFormulation::where(
481 [
482 'year' => $formulated_year,
483 'budget_specific_action_id' => $sp_acc->id,
484 'assigned' => true
485 ]
486 )->first() : '';
487
488 if (str_contains($sp_acc->specificable_type, 'BudgetProject') && !is_null($filter)) {
489 array_push($dataProjects['children'], [
490 'id' => $sp_acc->id,
491 'text' => "{$sp_acc->specificable->code} - {$sp_acc->code} | {$sp_acc->name}"
492 ]);
493 } elseif (str_contains($sp_acc->specificable_type, 'BudgetCentralizedAction') && !is_null($filter)) {
494 array_push($dataCentralizedActions['children'], [
495 'id' => $sp_acc->id,
496 'text' => "{$sp_acc->specificable->code} - {$sp_acc->code} | {$sp_acc->name}"
497 ]);
498 } elseif (!is_null($formulated) && $formulated && is_null($filter) && !$withoutFormulations) {
499 $withoutFormulations = true;
500 array_push($data, ['id' => '', 'text' => 'Sin formulaciones registradas', 'children' => []]);
501 }
502 }
503
504 /* Si el grupo Proyectos contiene registros los agrega a la lista */
505 if (count($dataProjects['children']) > 0) {
506 array_push($data, $dataProjects);
507 }
508 /* Si el grupo Acciones Centralizadas contiene registros los agrega a la lista */
509 if (count($dataCentralizedActions['children']) > 0) {
510 array_push($data, $dataCentralizedActions);
511 }
512
513 /* Si existen proyectos o acciones centralizadas y el arreglo tiene el texto 'sin formulaciones' se elimina del arreglo */
514 if (($key = array_search('Sin formulaciones registradas', array_column($data, 'text'))) !== false) {
515 array_splice($data, $key, 1);
516 }
517
518 return response()->json($data);
519 }

548 public function getOpenedAccounts($specificActionId, $selDate)
549 {
550 list($year, $month, $day) = explode("-", $selDate);
551 $records = [['id' => '', 'text' => 'Seleccione...']];
552 $anulationStatus = DocumentStatus::where('action', 'AN')->first();
553
554 $accounts = [];
555
556 $openedAccounts = BudgetAccountOpen::query()
557 ->with([
558 'budgetAccount',
559 'subSpecificFormulation'

562 $q->where('specific', '!=', '00');
563 })
564 ->whereHas('subSpecificFormulation', function ($q) use ($year, $specificActionId) {
565 $q
566 ->where('year', $year)
567 ->whereHas('specificAction', function ($qq) use ($specificActionId) {
568 $qq->where('id', $specificActionId);
569 });
570 })
571 ->get();
572
573 foreach ($openedAccounts as $account) {
574 if (!array_key_exists($account->budget_account_id, $accounts)) {
575 $accounts[$account->budget_account_id]['id'] = $account->budget_account_id;
576 $accounts[$account->budget_account_id]['amount'] = $account->total_year_amount;
577 $accounts[$account->budget_account_id]['budgetAccount'] = $account->budgetAccount;
578 $accounts[$account->budget_account_id]['currency'] = $account->subSpecificFormulation->currency;
579 } else {

581 }
582 }
583
584 $modificationAccounts = BudgetModificationAccount::query()
585 ->with(['budgetAccount', 'budgetSubSpecificFormulation.specificAction'])
586 ->whereHas('budgetSubSpecificFormulation', function ($query) use ($specificActionId, $year) {
587 $query
588 ->where('year', $year)
589 ->whereHas('specificAction', function ($q) use ($specificActionId) {
590 $q->where('id', $specificActionId);
591 });
592 })
593 ->whereHas('budgetAccount', function ($qq) {
594 $qq->where('specific', '!=', '00');
595 })
596 ->get();
597
598 foreach ($modificationAccounts as $account) {
599 if ($account->operation == 'I') {
600 if (!array_key_exists($account->budget_account_id, $accounts)) {
601 $accounts[$account->budget_account_id]['id'] = $account->budget_account_id;
602 $accounts[$account->budget_account_id]['amount'] = $account->amount;
603 $accounts[$account->budget_account_id]['budgetAccount'] = $account->budgetAccount;
604 $accounts[$account->budget_account_id]['currency'] = $account->budgetSubSpecificFormulation->currency;
605 } else {
606 $accounts[$account->budget_account_id]['amount'] += $account->amount;
607 }
608 } elseif ($account->operation == 'D') {
609 if (array_key_exists($account->budget_account_id, $accounts)) {
610 $accounts[$account->budget_account_id]['amount'] -= $account->amount;
611 }
612 }
613 }
614
615 $compromiseAccounts = BudgetCompromiseDetail::query()
616 ->whereHas('budgetSubSpecificFormulation', function ($query) use ($specificActionId, $year) {
617 $query
618 ->where('year', $year)
619 ->whereHas('specificAction', function ($q) use ($specificActionId) {
620 $q->where('id', $specificActionId);
621 });
622 })
623 ->whereHas('budgetAccount', function ($qq) {
624 $qq->where('specific', '!=', '00');
625 })
626 ->toBase()
627 ->get();
628
629 foreach ($compromiseAccounts as $account) {
630 if (
631 array_key_exists($account->budget_account_id, $accounts) &&
632 $account->document_status_id != $anulationStatus->id
633 ) {
634 $accounts[$account->budget_account_id]['amount'] -= $account->amount;
