145 if (is_null($document)) {
146 return response()->json([

149 }
150
151 // Ruta del archivo a eliminar
152 $file = $document->file;
153
154 DB::transaction(function () use ($document, $file, $request) {
155 if ($request->force_delete) {
156 $document->forceDelete();
157 if (Storage::disk((isset($request->store)) ? $request->store : 'documents')->exists($file)) {
158 Storage::disk((isset($request->store)) ? $request->store : 'documents')->delete($file);
159 }
160 } else {
161 $document->delete();
162 }
163 });
164 return response()->json(['result' => true, 'message' => 'Success'], 200);
