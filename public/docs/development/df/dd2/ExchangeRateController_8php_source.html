216 private function checkAlreadyExchange($request)
217 {
218 $alreadyExchange = ExchangeRate::where([
219 'from_currency_id' => $request->from_currency_id,
220 'to_currency_id' => $request->to_currency_id
221 ])->orderBy('id', 'desc')->first();
222 if ($alreadyExchange) {
223 $update = ['active' => false];
224 if ($alreadyExchange->end_at) {
225 $this->validate($request, [
226 'start_at' => ['after_or_equal:' . $alreadyExchange->end_at]
227 ], [
228 'start_at.after_or_equal' => __(
229 'La fecha de inicio debe ser mayor o igual a la fecha final ' .
230 'del último tipo de cambio registrado con estas monedas'
231 )
232 ]);
233 } else {
234 $this->validate($request, [
235 'start_at' => ['after_or_equal:' . $alreadyExchange->start_at]
236 ], [
237 'start_at.after_or_equal' => __(
238 'La fecha de inicio debe ser mayor o igual a la fecha de inicio ' .
239 'del último tipo de cambio registrado con estas monedas'
240 )
241 ]);
242 }
243
244 if ($request->active) {
245 $currentExchanges = ExchangeRate::where([
246 'from_currency_id' => $request->from_currency_id,
247 'to_currency_id' => $request->to_currency_id
248 ])->get();
249 foreach ($currentExchanges as $exchange) {
250 if (!$exchange->end_at) {
251 $update['end_at'] = $request->start_at;
252 }
253 try {
254 $exchange->update($update);
255 } catch (\Exception $e) {
256 Log::error($e->getMessage());
