79 /**

88 public function store(Request $request)
89 {
90 try {
91 DB::transaction(function () use ($request) {
92 $findAssets = Asset::query()
93 ->with(['assetSpecificCategory', 'assetDepreciationAssets', 'assetAdjustmentAssets.assetBook'])
94 ->has('assetSpecificCategory')
95 ->get();
96
97 $assets = [];
98 foreach ($findAssets as $asset) {
99 if (array_key_exists('depresciation_years', $asset->asset_details)) {
100 $depreciatedYears = $asset->asset_details['depresciation_years'];
101 $acquisitionDate = $asset->acquisition_date;
102 $acquisitionDate = new Carbon($acquisitionDate);
103 $currentDate = Carbon::now();
104 if ($acquisitionDate->addYears($depreciatedYears) >= $currentDate) {
105 array_push($assets, $asset);
106 }
107 }
108 }
109
110 $assetMethod = AssetDepreciationMethod::query()
111 ->where('depreciation_type_id', 1)
112 ->first();
113
114 $currentFiscalYear = FiscalYear::query()
115 ->select('year')
116 ->where(['active' => true, 'closed' => false])
117 ->orderBy('year', 'desc')
118 ->first();
119
120 $defaultCurrency = Currency::query()
121 ->where('default', true)
122 ->first();
123
124 $depreciationDate = $currentFiscalYear->year . '-12-31';
125 $formatedDepreciationDate = date('d/m/Y', strtotime($depreciationDate));
126
127 $arr = [];
128 $totalDepreciationAmount = 0.00;
129
130 foreach ($assets as $key => $asset) {
131 $exchangeRate = ExchangeRate::query()
132 ->where('active', true)
133 ->where('start_at', '<=', $depreciationDate)
134 ->where(function ($query) use ($depreciationDate) {
135 $query
136 ->where('end_at', '>=', $depreciationDate)
137 ->orWhereNull('end_at');
138 })
139 ->where('from_currency_id', $defaultCurrency->id)
140 ->where('to_currency_id', $asset->currency_id)
141 ->orderBy('id', 'desc')
142 ->first();
143
144 if (!$exchangeRate && $asset->currency_id != $defaultCurrency->id) {
145 throw new \Exception(
146 'No se puede realizar la depreciación, por favor revise los tipos de' .
147 'cambios activos para la fecha de depreciación ' . $formatedDepreciationDate .
148 ':asset.depreciation.index',
149 1
150 );
151 }
152
153 $details = $asset->asset_details;
154
155 $cond_a = (!empty($details['acquisition_value']) && $details['acquisition_value'] != 0);
156 $cond_b = (!empty($details['depresciation_years']) && $details['depresciation_years'] != 0);
157 $cond_c = $asset->acquisition_date != null;
158
159 if ($cond_a && $cond_b && $cond_c) {
160 if ($assetMethod) {
161 $depreciationFormula = $assetMethod->depreciation_type_id->getTranslateFormula();
162 $months = 12;
163 $days = 30;
164 $totalDays = 360;
165 $notDepreciated = false;
166 $depreciatedYears = $asset->assetDepreciationAssets->last()->depreciated_years ?? 0;
167
168 // Toma los valores del bien
169 $fisrtAcquisitionValue = $details['acquisition_value'];
170 $residualValue = $details['residual_value'];
171 $depresciationYears = $details['depresciation_years'];
172
173 if (count($asset->assetAdjustmentAssets) > 0) {
174 $fisrtAcquisitionValue = $asset->assetAdjustmentAssets->last()->assetBook->amount;
175 $residualValue = $asset->assetAdjustmentAssets->last()->residual_value;
176 $depresciationYears = $asset->assetAdjustmentAssets->last()->depresciation_years;
177 }
178
179 if (
180 count($asset->assetDepreciationAssets) > 0 &&
181 ($depreciatedYears + 1) > $depresciationYears &&
182 $asset->assetDepreciationAssets->last()->days_remaining == 0
183 ) {
184 // Indica que el bien ya no se puede depreciar
185 $notDepreciated = true;
186 }
187
188 if ($notDepreciated === false) {
189 $depreciationFormula = str_replace(
190 'acquisition_value',
191 currency_format($fisrtAcquisitionValue, 2, true),
192 $depreciationFormula
193 );
194
195 $depreciationFormula = str_replace(
196 'residual_value',
197 currency_format($residualValue ?? 0, 2, true),
198 $depreciationFormula
199 );
200
201 $depreciationFormula = str_replace(
202 'depresciation_years',
203 currency_format($depresciationYears, 2, true),
204 $depreciationFormula
205 );
206
207 // Se calcula el resultado de la formula de depreciación
208 $string = 'select(' . $depreciationFormula . ')';
209 $calc = DB::select(DB::raw($string));
210 $col = '?column?';
211 $value = $calc[0]->$col ?? $calc[0]->case;
212
213 if ($exchangeRate && $asset->currency_id != $defaultCurrency->id) {
214 $value = $value * $exchangeRate->amount;
215 }
216
217 // Se indica la fecha del periodo a tomar según el caso correspondiente
218 $useRemainingDays = false;
219
220 if (count($asset->assetDepreciationAssets) == 0) {
221 // Toma la fecha de adquisición del bien y la divide en año, mes y día
222 $date = explode('-', $asset->acquisition_date);
223 } elseif (
224 count($asset->assetDepreciationAssets) > 0 &&
225 ($depreciatedYears + 1) <= $depresciationYears
226 ) {
227 // Toma la fecha de adquisición del bien y la divide en año, mes y día
228 $date = explode('-', '2023-01-01');
229 } elseif (
230 count($asset->assetDepreciationAssets) > 0 &&
231 ($depreciatedYears + 1) > $depresciationYears &&
232 $asset->assetDepreciationAssets->last()->days_remaining > 0
233 ) {
234 // Usa los días restantes para depreciar
235 $useRemainingDays = true;
236 }
237
238 // Calcula el valor diario de adquisición del bien
239 $dailyAcquisitionValue = ($value / 12) / 30;
240
241 if ($useRemainingDays == false) {
242 // Calcula el valor de los meses pendientes para depreciar
243 $pendingMonths = $months - $date[1];
244
245 // Calcula el valor de los días pendientes para depreciar
246 $pendingDays = ($days - $date[2]) + 1;
247
248 // Calcula el valor de los meses pendientes para depreciar en días
249 $pendingMonthsToDays = $pendingMonths * $days;
250
251 // Calcula el valor del total de días pendientes para depreciar
252 $totalDepreciationDays = $pendingDays + $pendingMonthsToDays;
253
254 // Calcula el valor de los días que quedarón pendientes para el final de la depreciación
255 $pendingDaysAfterDepreciation = $asset->assetDepreciationAssets->last() &&
256 $asset->assetDepreciationAssets->last()->days_remaining > 0 ?
257 $asset->assetDepreciationAssets->last()->days_remaining :
258 $totalDays - $totalDepreciationDays;
259
260 $value = $dailyAcquisitionValue * $totalDepreciationDays;
261 } else {
262 $value = $dailyAcquisitionValue *
263 $asset
264 ->assetDepreciationAssets
265 ->last()
266 ->days_remaining;
267 $pendingDaysAfterDepreciation = 0;
268 }
269
270 $depreciatedYears = $depreciatedYears + 1;
271 $totalDepreciationAmount = $totalDepreciationAmount + $value;
272
273 $arr[] = [
274 'asset_id' => $asset->id,
275 'name' => $asset['assetSpecificCategory']['name'],
276 'amount' => $value,
277 'depreciated_years' => $depreciatedYears,
278 'remaining_days' => $pendingDaysAfterDepreciation
279 ];
280 }
281 } else {
282 throw new \Exception(
283 'Debe configurar previamente el método de depreciación a utilizar:asset.setting.index',
284 1
285 );
286 }
287 }
288 }
289
290 $documentStatus = DocumentStatus::where('action', 'PR')->first();
291
292 $codeSetting = CodeSetting::where('table', 'asset_depreciations')->first();
293 if (is_null($codeSetting)) {
294 throw new \Exception(
295 'Debe configurar previamente el formato para el código a generar:asset.setting.index',
296 1
297 );
298 }
299
300 $documentStatusCancel = DocumentStatus::where('action', 'AN')->first();
301 $oldAssetDepreciations = AssetDepreciation::query()
302 ->where('document_status_id', '!=', $documentStatusCancel->id)
303 ->where('year', $currentFiscalYear->year)
304 ->get();
305
306 if (count($oldAssetDepreciations) > 0) {
307 throw new \Exception(
308 'Solo se puede realizar una depreciación por año fiscal:asset.depreciation.index',
309 1
310 );
311 }
312
313 $code = generate_registration_code(
314 $codeSetting->format_prefix,
315 strlen($codeSetting->format_digits),
316 (strlen($codeSetting->format_year) == 2) ? (isset($currentFiscalYear) ?
317 substr($currentFiscalYear->year, 2, 2) : date('y')) : (isset($currentFiscalYear) ?
318 $currentFiscalYear->year : date('Y')),
319 AssetDepreciation::class,
320 $codeSetting->field
321 );
322
323 if (isset(auth()->user()->profile) && isset(auth()->user()->profile->institution_id)) {
324 $institution = Institution::where(['id' => auth()->user()->profile->institution_id])->first();
325 } else {
326 $institution = Institution::where(['active' => true, 'default' => true])->first();
327 }
328
329 $assetDepreciation = AssetDepreciation::create([
330 'code' => $code,
331 'year' => $currentFiscalYear->year,
332 'amount' => $totalDepreciationAmount,
333 'document_status_id' => $documentStatus->id,
334 'institution_id' => $institution->id
335 ]);
336
337 foreach ($arr as $item) {
338 $lastAssetBook = AssetBook::query()
339 ->where('asset_id', $item['asset_id'])
340 ->toBase()
341 ->get()
342 ->last();
343
344 $amount = ($lastAssetBook?->amount - $item['amount']) < 0 ?
345 ($lastAssetBook?->amount - $item['amount']) * -1 :
346 ($lastAssetBook?->amount - $item['amount']);
347
348 $assetBook = AssetBook::create([
349 'asset_id' => $item['asset_id'],
350 'amount' => $amount,
351 ]);
352
353 AssetDepreciationAsset::create([
354 'asset_depreciation_id' => $assetDepreciation->id,
355 'asset_id' => $item['asset_id'],
356 'asset_book_id' => $assetBook->id,
357 'amount' => $item['amount'],
358 'depreciated_years' => $item['depreciated_years'],
359 'days_remaining' => $item['remaining_days']
360 ]);
361 }
362 });
363
364 $request->session()->flash('message', ['type' => 'store']);
365 return response()->json(['redirect' => route('asset.depreciation.index')], 200);
366 } catch (\Throwable $th) {
367 Log::error($th->getMessage());
368 $message = explode(':', $th->getMessage());
369 $request->session()->flash('message', [
370 'type' => 'other', 'title' => 'Alerta', 'icon' => 'screen-error', 'class' => 'growl-danger',
371 'text' => $message[0],
372 ]);
373
374 return response()->json(['result' => false, 'redirect' => route($message[1])], 200);
375 }
376 }

578 /**
