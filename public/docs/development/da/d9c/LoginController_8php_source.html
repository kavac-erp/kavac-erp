86 /**
87 * Gestiona una petición de acceso a la aplicación.
88 *
89 * @param Request $request
90 *
91 * @return RedirectResponse|Response|JsonResponse
92 *
93 * @throws ValidationException
94 */
95 public function login(Request $request)
96 {
97 $this->validateLogin($request);
98
99 // Objeto con información del usuario a autenticar
100 $user = User::where('username', $request->username)->first();
101
102 if ($user !== null) {
103 if (!is_null($user->blocked_at)) {
104 return $this->sendLockedAccountResponse($request);
105 } elseif (!$user->active) {
106 return ($this->sendInactiveAccountRequest($request));
107 } elseif (
108 $request->ip() !== '127.0.0.1' &&
109 Session::where('user_id', $user->id)->where('ip_address', '<>', $request->ip())->first()
110 ) {
111 return ($this->sendHasActiveSessionRequest($request));
112 }
113 }
114
115 if (method_exists($this, 'hasTooManyLoginAttempts') && $this->hasTooManyLoginAttempts($request)) {
116 // elimina la cantidad de intentos fallidos del usuario y se procede al bloqueo del mismo
117 $this->clearLoginAttempts($request);
118
119 $this->fireLockoutEvent($request);
120
121 return $this->sendLockoutResponse($request);
122 }
123
124 if ($this->attemptLogin($request)) {
125 return $this->sendLoginResponse($request);
126 }
127
128 $this->incrementLoginAttempts($request);
129
130 return $this->sendFailedLoginResponse($request);
131 }
