352 /**
353 * Método que permite agregar el contenido del reporte a generar
354 *
355 * @author Ing. Roldan Vargas <rvargas@cenditel.gob.ve> | <roldandvg@gmail.com>
356 *
357 * @param string $body Plantilla a utilizar para el reporte en caso de estar establecido
358 * como isHTML, en caso contrario será un texto a incluir en el cuerpo
359 * del reporte
360 * @param boolean $isHTML Establece si el cuerpo del reporte es una plantilla de blade a renderizar
361 * @param array $htmlParams Conjunto de parámetros requeridos por la plantilla de blade
362 */
363 public function setBody($body, $isHTML = true, $htmlParams = [])
364 {
365 /* Contenido del reporte */
366 $htmlContent = $body;
367 /* Configuración sobre el autor del reporte */
368 $this->pdf->SetAuthor('Sistema de Gestión de Recursos - ' . config('app.name'));
369 /* Configuración del título de reporte */
370 $this->pdf->SetTitle($this->title);
371 /* Configuración sobre el asunto del reporte */
372 $this->pdf->SetSubject($this->subject);
373 /* Configuración de los márgenes del cuerpo del reporte */
374 $this->pdf->SetMargins(7, 45, 7);
375 /* Establece si se configura o no las fuentes para sub configuraciones */
376 $this->pdf->SetFontSubsetting(false);
377 /* Configuración de la fuente por defecto del cuerpo del reporte */
378 $this->pdf->SetFontSize('10px');
379 /*
380 | Configuración que permite realizar un salto de página automático al alcanzar el límite inferior del cuerpo
381 | del reporte
382 */
383 $this->pdf->SetAutoPageBreak(true, 15); //PDF_MARGIN_BOTTOM
384 /** Agrega las respectivas páginas del reporte */
385 $this->pdf->AddPage($this->orientation, $this->format);
386
387 if ($isHTML) {
388 $view = View::make($body, $htmlParams);
389 $htmlContent = $view->render();
390 }
391 /* Escribre el contenido del reporte */
392 $this->pdf->writeHTML($htmlContent, true, false, true, false, '');
393 /* Establece el apuntador del reporte a la última página generada */
394 $this->pdf->lastPage();
395 /*
396 | Genera el reporte. Las opciones disponibles son:
397 |
398 | I: Genera el archivo directamente para ser visualizado en el navegador
399 | D: Genera el archivo y forza la descarga del mismo
400 | F: Guarda el archivo generado en la ruta del servidor establecida por defecto
401 | S: Devuelve el documento generado como una cadena de texto
402 | FI: Es equivalente a las opciones F + I
403 | FD: Es equivalente a las opciones F + D
404 | E: Devuelve el documento del tipo mime base64 para ser adjuntado en correos electrónicos
405 */
406 $this->pdf->Output(storage_path() . '/reports/' . $this->filename, 'F');
407 }
