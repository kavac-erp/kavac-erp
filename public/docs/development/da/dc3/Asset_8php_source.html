625  public function scopeCodeClasification($query, $type, $category, $subcategory, $specific, $is_dis, $ids)
626  {
627  if ($is_dis == false) {
628  if ($type != "") {
629  if ($category != "") {
630  if ($subcategory != "") {
631  if ($specific != "") {
632  return $query->where('asset_type_id', $type)
633  ->where('asset_category_id', $category)
634  ->where('asset_subcategory_id', $subcategory)
635  ->where('asset_specific_category_id', $specific);
636  }
637  return $query->where('asset_type_id', $type)
638  ->where('asset_category_id', $category)
639  ->where('asset_subcategory_id', $subcategory);
640  }
641  return $query->where('asset_type_id', $type)
642  ->where('asset_category_id', $category);
643  }
644  return $query->where('asset_type_id', $type);
645  } else {
646  return $query;
647  }
648  } else {
649  if ($type != "") {
650  if ($category != "") {
651  if ($subcategory != "") {
652  if ($specific != "") {
653  return $query->where('asset_type_id', $type)
654  ->where('asset_category_id', $category)
