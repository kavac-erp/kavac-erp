160
161 ]);
162 $data = array_merge($data, $payrollSocioeconomic['payrollChildrens']
163 ->map(function ($model) use ($payrollSocioeconomic) {
164 return [
165 'payroll_staff_id_number' => $payrollSocioeconomic->payrollStaff->id_number,
166 'marital_status' => '',
167 'first_name' => $model->first_name,
168 'last_name' => $model->last_name,
169 'payroll_relationship' => $model->payrollRelationship?->name,
170 'id_number' => $model->id_number,
171 'birthdate' => $model->birthdate,
172 'address' => $model->address,
173 'gender' => $model->payrollGender?->name,
174 'disability' => $model->has_disability ? $model->payrollDisability?->name : [],
175 'student' => $model->is_student ? [
176 'study_center' => $model->study_center,
177 'payroll_schooling_level' => $model->payrollSchoolingLevel?->name,
178 'scholarships' => $model->has_scholarships
179 ? ['scholarship_type' => $model->payrollScholarshipType?->name]
180 : [],
181 ] : []
182
183 ];
184 })->toArray());
185 });
186
187 return [
188 'Datos Socioeconomicos' => new SocioStaffExportFromButton(collect($data)),
189 'validation' => new PayrollSocioeconomicValidationExport(),
