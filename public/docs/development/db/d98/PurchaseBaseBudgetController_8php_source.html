288 if ($record) {
289 foreach (PurchaseRequirement::where('purchase_base_budget_id', $id)->orderBy('id', 'ASC')->get() as $r) {
290 $r->delete();
291 }

293 PurchasePivotModelsToRequirementItem::where('relatable_id', $id)
294 ->orderBy('id', 'ASC')->get() as $r
295 ) {
296 $r->delete();
297 }
298 $record->delete();
299 } else {
300 return response()->json(['message' => 'El registro ya fue eliminado.'], 200);
301 }
302 return response()->json(['message' => 'success'], 200);
