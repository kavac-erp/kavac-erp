165  * Método que valida la estructura del archivo usado para realizar la conciliación bancaria

168  *
169  * @param [stream] $file
170  *
171  * @return array|object
172  */
173  public function validateFileStructure($file, $bank_id)
174  {
175  // Encontrar el archivo de configuración para el banco asociado a la cuenta bancaria
176  $fileStructure = $this->getFileValidationArray($bank_id);
177 
178  // Si no existe, lanzar error de validación
179  throw_if(
180  is_null($fileStructure),
181  'El banco asociado a la cuenta bancaria no tiene una configuración para el archivo ' .
182  'de conciliación bancaria. Por favor registre una configuración.'
183  );
184 
185  // Leer archivo csv/.txt
186  $filePath = $file->getRealPath();
187 
188  // Delimitador del archivo csv/.txt
189  $delimiter = $this->guessDelimiter($filePath);
190  // Obtener el string correspodiente a los delimitadores ';' | ',' | '\t'
191  $translatedDelimiter = $this->getFileConfigurationDelimiter($delimiter);
192 
193  // Si el delimitador del archivo csv/.txt no coincide con el del archivo de configuración, lanzar error de validación
194  throw_if(
195  $translatedDelimiter != $fileStructure['separated_by'],
196  'El archivo no se encuentra separado por el separador indicado en la configuración'
197  );
198 
199 
200  $csvData = [];
201 
202 
203  if (($open = fopen($filePath, "r")) !== false) {
204  if ($fileStructure["read_start_line"] == true) {
205  fgetcsv($open, 700, $delimiter);
206  }
207 
208  while (($data = fgetcsv($open, 700, $delimiter)) !== false) {
209  $csvData[] = is_array($data) ? trim($data[0]) : trim($data);
210  }
211 
212  if ($fileStructure["read_end_line"] == false) {
213  array_pop($csvData);
214  }
215 
216  fclose($open);
217  }
218 
219  $decimalSeparator = $this->guessDecimalSeparator($csvData);
220  $translatedDecimalSeparator = $this->getFileConfigurationDecimalSeparator($decimalSeparator);
221 
222  throw_if(
223  $translatedDecimalSeparator != $fileStructure['decimal_separator'],
224  'El archivo no esta usando el separador de decimales indicado en la configuración'
225  );
226  return $csvData;
227  }
