375 /* Se guarda el archivo y se almacena */

381 );

385 'document_id' => $upDoc->getDocStored()->id,
386 'required_document_id' => $request->reqDocs[$key],
387 ]);

389 }
390 }
391 }
392 session()->flash('message', ['type' => 'store']);
393 return redirect()->route('purchase.suppliers.index');
