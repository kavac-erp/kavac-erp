140 {
141 $fiscalYear = $data['fiscal_year'];
142 $inicioAnho = Carbon::createFromDate($fiscalYear)->startOfYear()->format('Y-m-d');
143 $finAnho = Carbon::createFromDate($fiscalYear)->endOfYear()->format('Y-m-d');
144 $payrollStaffs = $encoded
145 ? (json_decode($data['payroll_staffs'] ?? '') ?? [])

148

153 return $this->getQueryBase(

155 $finAnho,
156 $nameResponsibleArc,
157 $rifResponsibleArc
158 )
159 ->when(
160 !in_array('todos', $payrollStaffs),
161 fn ($query) => $query->whereIn('id', $payrollStaffs)
162 )
163 ->get()->map(function ($resource) use ($arcConceptIds) {
164 $dataPayrolls = $resource->payrollStaffPayrolls->map(function ($item) use ($arcConceptIds) {
165 $assignmentsTotal = 0;
166 $deductionsTotal = 0;
167
168 foreach ($item->concept_type['Asignaciones'] as $concept) {
169 if (in_array($concept['id'], $arcConceptIds)) {
170 $assignmentsTotal += $concept['value'];

174 foreach ($item->concept_type['Deducciones'] as $concept) {

176 $deductionsTotal += $concept['value'];
177 }
178 }
179
180 return [
181 'payroll_id' => $item->payroll_id,
182 'payroll_payment_period' => Carbon::parse($item->payroll->payrollPaymentPeriod->end_date)->format('m'),
183 'total_value' => $assignmentsTotal - $deductionsTotal,
184 ];
185 });
186
187 $groupedData = $dataPayrolls->groupBy('payroll_payment_period')->map(function ($items, $month) {
188 $totalValue = $items->sum('total_value');

190 return [

193 ];
194 })->sortBy('month');
195
196 $institution = $resource->payrollEmployment?->department?->institution ?? null;
197 if (isset($institution) && Str::length($institution->rif) == 10) {

200
201 return [
202 'id' => $resource->id,
203 'payroll_staff' => [

213 'city' => $resource->payrollEmployment?->department?->institution?->city?->name ?? '',

216 'po_box' => '',
217 'phone' => '',
218 ],

221 'name' => $resource->name_arc_responsible,

227 'arc' => $groupedData->sum('total_value'),
