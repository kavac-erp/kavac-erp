85  $model = FinancePaymentExecute::class;
86  }
87 
88  $codeSetting = CodeSetting::where([
89  'module' => 'finance',

91  'field' => $field
92  ])->first();
93 
94  if (!isset($codeSetting)) {
95  $codeSetting = CodeSetting::create([
96  'module' => 'finance',
97  'table' => 'finance_' . $table,
98  'field' => $field,
99  'format_prefix' => $prefix,
100  'format_digits' => $digits,
101  'format_year' => $sufix,
102  'model' => $model
103  ]);
104  }
105 
106  /** @var boolean $saved Define el estado verdadero para indicar que se ha registrado información */
107  $saved = true;
108  }
109  }
