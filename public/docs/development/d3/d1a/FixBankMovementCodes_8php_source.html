230 * @param string $model Namespace y nombre del modelo en donde se aplicará el nuevo código
231 * @param string $field Nombre del campo del código a generar

233 *
234 * @return string|array Retorna una cadena con el nuevo código

236 private function customGenerateRegistrationCodeForBankMovements($prefix, $code_length, $year, $model, $field, $order_field)
237 {
238 $newCode = 1;
239
240 $targetModel = $model::select('id', $field, $order_field)->where($field, 'like', "{$prefix}-%-{$year}")
241 ->withTrashed()->orderBy($order_field, 'desc')->orderBy('id', 'desc')->first();
242
243 $newCode += ($targetModel) ? (int)explode('-', $targetModel->$field)[1] : 0;
244
245 if (strlen((string)$newCode) > $code_length) {
246 return ["error" => "El nuevo código excede la longitud permitida"];
247 }
248
249 return "{$prefix}-{$newCode}-{$year}";
250 }
