355 public function approveBudgetaryAvailability(Request $request)
356 {
357 try {
358 if ($request->module == 'Purchase') {
359 $PurchaseBaseBudget = PurchaseBaseBudget::query()->where('id', $request->id)->firstOrFail();
360 $BudgetaryAvailable = PurchaseBudgetaryAvailability::query()
361 ->where('purchase_base_budgets_id', $PurchaseBaseBudget->id)->get();
362
363 if ($PurchaseBaseBudget->availability == 'Disponible') {
364 DB::transaction(
365 function () use ($BudgetaryAvailable, $PurchaseBaseBudget, $request) {
366 foreach ($BudgetaryAvailable as $budgetaryAvailable) {
367 //cambiar el estatus a aprobado
368 $budgetaryAvailable['availability'] = 2;
369 $budgetaryAvailable->save();
370 }
371 }
372 );
373 return response()->json(['message' => 'Success'], 200);
374 } elseif ($PurchaseBaseBudget->availability == 'AP') {
375 new \Exception('Esta disponibilidad presupuestaria ya fue aprobada');
376 } elseif ($PurchaseBaseBudget->availability == 'No_Disponible') {
377 new \Exception("Esta disponibilidad presupuestaria no puede ser aprobada. Su estatus es 'No Disponible'.");
378 } else {
379 new \Exception('Esta disponibilidad presupuestaria no puede ser aprobada, debido a que su estatus es: ' . $PurchaseBaseBudget->available);
380 }
381 } elseif ($request->module == 'Payroll' && Module::has('Payroll') && Module::isEnabled('Payroll')) {
382 $BudgetaryAvailable = \Modules\Payroll\Models\Payroll::query()
383 ->where('id', $request->id)->firstOrFail();
384 $payrollPaymentPeriod = $BudgetaryAvailable->payrollPaymentPeriod;
385
386 if ($payrollPaymentPeriod) {
387 if ($payrollPaymentPeriod->availability_status == 'AP') {
388 new \Exception('Esta disponibilidad presupuestaria ya fue aprobada');
389 } elseif ($payrollPaymentPeriod->availability_status == 'available') {
390 DB::transaction(
391 function () use ($BudgetaryAvailable) {
392 //cambiar el estatus a aprobado
393 $BudgetaryAvailable->payrollPaymentPeriod->availability_status = 'AP';
394 $BudgetaryAvailable->payrollPaymentPeriod->save();
395 }
396 );
397 return response()->json(['message' => 'Success'], 200);
398 }
399 } else {
400 new \Exception('Esta disponibilidad presupuestaria no puede ser aprobada');
401 }
402 }
403 } catch (\Exception $e) {
404 Log::error($e->getMessage());
405 $message = str_replace("\n", "", $e->getMessage());
406 if (strpos($message, 'ERROR') !== false && strpos($message, 'DETAIL') !== false) {
407 $pattern = '/ERROR:(.*?)DETAIL/';
408 preg_match($pattern, $message, $matches);
409 $errorMessage = trim($matches[1]);
410 } else {
411 $errorMessage = $message;
412 }
413
414 return response()->json(
415 ['message' => [
416 'type' => 'other',
417 'title' => 'Alerta',
418 'icon' => 'screen-error',
419 'class' => 'growl-danger',
420 'text' => 'No se pudo completar la operación. ' . ucfirst($errorMessage)
421 ]
422 ],
423 500
424 );
425 }
