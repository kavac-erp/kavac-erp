69 /**
70 * Ejecuta la acción necesaria para restaurar los datos a partir de un respaldo
71 *
72 * @author Ing. Roldan Vargas <rvargas@cenditel.gob.ve> | <roldandvg@gmail.com>
73 *
74 * @param string $filename Nombre del archivo con el respaldo a ser restaurado
75 * @param object $request Objeto con información de la petición
76 *
77 * @return boolean Devuelve verdadero si se realizó la restauración de la base de datos,
78 * de lo contrario devuelve falso
79 */
80 public function restore($filename, $request)
81 {
82 // Ruta en la que se realiza el respaldo de la base de datos
83 $path = Storage::disk(config('backup.backup.destination.disks')[0])->getAdapter()->getPathPrefix();
84 // Instancia a la clase
85 $zip = new ZipArchive();
86 // Objeto con información del archivo a descomprimir
87 $res = $zip->open($path . config('app.name') . "/" . $filename);
88 if ($res === true) {
89 try {
90 $request->session()->regenerate();
91 // Nombre del archivo a descomprimir
92 $snapName = str_replace(".zip", "", $filename);
93 $zip->renameName('db-dumps/postgresql-kavac.sql', $snapName . '.sql');
94 $zip->extractTo($path);
95 $zip->close();
96 // Código de respuesta en la restauración de datos
97 $exitCode = Artisan::call('snapshot:load ' . $snapName);
98 // Datos de respuesta en la ejecución del comando
99 $output = Artisan::output();
100 // Registra un log con los resultados de la ejecución del comando
101 Log::info(
102 "Backpack\BackupManager -- " .
103 'se ha generado una nueva restauración de la base de datos desde la interfaz administrativa' .
104 ". \r\n" .
105 $output
106 );
107 return (strpos($output, 'loaded') > 0);
108 } catch (Exception $e) {
109 // Registra un log de la excepción
110 Log::error($e->getMessage());
111 return false;
112 }
113 }
114
115 return false;
116 }

118 /**
119 * Muestra el listado de archivos de respaldo
120 *
121 * @author Ing. Roldan Vargas <rvargas@cenditel.gob.ve> | <roldandvg@gmail.com>
122 *
123 * @param string $disk Nombre del disco del sistema de archivos
124 * @param string $dir Nombre del directorio del sistema de archivos
125 *
126 * @return array Devuelve un arreglo con el listado de archivos de respaldo en orden descendente
127 */
128 public function getList($disk, $dir)
129 {
130 // Objeto con información del disco a usar para obtener el listado de archivos disponibles
131 $storage_disk = Storage::disk($disk[0]);
132 // Objeto con información de los archivos disponibles
133 $files = $storage_disk->files($dir);
134 // Listado con información detallada de los archivos a restaurar
135 $backups = [];
136 // Genera un arreglo con la información detallada de los archivos de respaldo (tamaño y fecha de creación)
137 foreach ($files as $k => $f) {
138 if (substr($f, -4) == '.zip' && $storage_disk->exists($f)) {
139 $backups[] = [
140 'file_path' => $f,
141 'file_name' => str_replace($dir . '/', '', $f),
142 'file_size' => $this->humanFileSize($storage_disk->size($f)),
143 'last_modified' => $storage_disk->lastModified($f),
144 ];
145 }
146 }
147 // Reordena el arreglo para mostrar los archivos de respaldo desde el mas nuevo al mas antiguo
148 return array_reverse($backups);
149 }

211 * @return string

213 public function humanFileSize($size, $precision = 2)
214 {
215 // Listado de unidades de capacidad
216 $units = ['B','kB','MB','GB','TB','PB','EB','ZB','YB'];
217 // base de cálculo para determinar el tamaño en formato legible
218 $step = 1024;
219 // Contador
220 $i = 0;
221 while (($size / $step) > 0.9) {
222 // Tamaño del archivo
223 $size = $size / $step;
224 $i++;
225 }
226 return round($size, $precision) . $units[$i];
