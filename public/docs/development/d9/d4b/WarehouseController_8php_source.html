125 * Valida y Registra un nuevo almacén
126 *

133 public function store(Request $request)
134 {
135 $this->validate($request, $this->validateRules, $this->messages);
136
137 $warehouse = Warehouse::create([
138 'name' => $request->input('name'),
139 'address' => $request->input('address'),
140 'parish_id' => $request->input('parish_id'),
141 'active' => !empty($request->active) ? $request->input('active') : false,
142 ]);
143 if (empty($request->institution_id)) {
144 $institution = Institution::where('active', true)->where('default', true)->first();
145 }
146 $institution_id = empty($request->institution_id) ? $institution->id : $request->institution_id;
147
148 $warehouse_institution = WarehouseInstitutionWarehouse::create([
149 'institution_id' => $institution_id,
150 'warehouse_id' => $warehouse->id,
151 'main' => !empty($request->main) ? $request->input('main') : false,
152 ]);
153
154 $paramMultiWarehouse = Parameter::where([
155 'active' => true, 'required_by' => 'warehouse',
156 'p_key' => 'multi_warehouse', 'p_value' => 'true'
157 ])->first();
158
159 if (is_null($paramMultiWarehouse) || ($paramMultiWarehouse->p_value == false)) {
160 $inst_wares = WarehouseInstitutionWarehouse::where('institution_id', $institution_id)
161 ->with('warehouse')->get();
162
163 foreach ($inst_wares as $inst_ware) {
164 if ($inst_ware->warehouse_id != $warehouse->id) {
165 $record = Warehouse::find($inst_ware->warehouse_id);
166 $record->active = ($warehouse->active == true) ? false : $record->active;
167 $record->save();
168
169 if (!empty($request->main) && ($inst_ware->main == $request->main)) {
170 $inst_ware->main = !$inst_ware->main;
171 $inst_ware->save();
172 }
173 }
174 }
175 }
176
177 return response()->json(['record' => $warehouse, 'message' => 'Success'], 200);
178 }
