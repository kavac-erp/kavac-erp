498 /**
499 * Filtro de registros
500 *
501 * @author Francisco J. P. Ruiz <fjpenya@cenditel.gob.ve>
502 *
503 * @param Request $request Datos de la petición
504 * @param integer $perPage Número de elementos por página
505 * @param integer $page Página de la consulta
506 *
507 * @return JsonResponse
508 */
509 public function filterRecords(Request $request, $perPage = 10, $page = 1)
510 {
511 $user = auth()->user();
512 $user_profile = Profile::with('institution')->where('user_id', $user->id)->first();
513 $institution_id = null;
514
515 if ($user->isAdmin() && $request->institution) {
516 $institution_id = $request->institution;
517 } elseif ($user_profile && $user_profile->institution && $user_profile->institution->id == $request->institution) {
518 $institution_id = $request->institution;
519 }
520
521 $documentStatusEL = default_document_status_el();
522
523 $query = AccountingEntry::query()
524 ->withCount("pivotEntryable")
525 ->where('document_status_id', '!=', $documentStatusEL->id)
526 ->whereHas('documentStatus', function ($query) use ($documentStatusEL) {
527 $query->where('id', '!=', $documentStatusEL->id);
528 });
529
530 $search = $request->search ?: $request->reference;
531
532 if ($request->typeSearch == 'reference' || $request->typeSearch == 'origin') {
533 $query->where(function ($q) use ($search) {
534 $q->where('reference', 'like', "%{$search}%")
535 ->orWhere('from_date', 'like', "%{$search}%")
536 ->orWhere('concept', 'like', "%{$search}%");
537 });
538
539 if ($institution_id) {
540 $query->where('institution_id', $institution_id);
541 }
542
543 if ($request->typeSearch == 'origin' && $request->category) {
544 $query->where('accounting_entry_category_id', $request->category);
545 }
546 }
547
548 if ($request->filterDate == 'generic') {
549 if ($request->year) {
550 $query->whereYear('from_date', $request->year);
551 }
552 if ($request->month) {
553 $query->whereMonth('from_date', $request->month);
554 }
555 } else {
556 $query->whereBetween('from_date', [$request->init, $request->end]);
557 }
558
559 $closeFiscalYear = FiscalYear::query()->get();
560 $entriesId = $closeFiscalYear->flatMap(function ($fYear) {
561 return $fYear->entries ? collect($fYear->entries)->pluck('id') : collect();
562 })->filter()->toArray();
563
564 $query->whereNotIn('id', $entriesId);
565
566 $total = $query->count();
567 $records = $query->orderBy('approved', 'ASC')
568 ->orderBy('id', 'ASC')
569 ->orderBy('from_date', 'ASC')
570 ->orderBy('reference', 'ASC')
571 ->offset(($page - 1) * $perPage)
572 ->limit($perPage)
573 ->get();
574
575 foreach ($records as $record) {
576 $record->concept = strip_tags($record->concept);
577 }
578
579 $lastPage = max((int) ceil($total / $perPage), 1);
580
581 return response()->json([
582 'records' => $records,
583 'total' => $total,
584 'lastPage' => $lastPage,
585 ], 200);
586 }

816 $accounts = [];
817
818 foreach ($request->objectsList as $data) {
819 foreach ($accounting_accounts as $account) {
820 if ($account->id == $data['account']) {
821 if (!array_key_exists($data['account'], $accounts)) {
822 $account->amount = $data['amount'];
823 $account->is_retention = $data['is_retention'];
824
825 if (isset($data['debit']) && $data['debit'] == true) {
826 $account->debit = $data['amount'];
827 } else {
828 $account->assets = $data['amount'];
829 }

831 $accounts[$data['account']] = $account;
832 } else {
833 $account->amount += $data['amount'];
834 $account->is_retention = $data['is_retention'];
835
836 if (isset($data['debit']) && $data['debit'] == true) {
837 $account->debit += $data['amount'];
838 } else {
839 $account->assets += $data['amount'];
840 }
841
842 $accounts[$data['account']] = $account;
843 }
844 }
845 }
846 }
847
848 $retAmount = 0;
849 $assetsLength = 0;
850
851 foreach ($accounts as $acc) {
852 if ($acc->is_retention == true) {
853 $retAmount += $acc->amount;
854 } elseif ($acc->is_retention == false && isset($acc->assets)) {
855 $assetsLength++;
856 }
857 }
858
859 if ($retAmount > 0) {
860 $retAmount = $retAmount / $assetsLength;
861 }
862
863 foreach ($accounts as $acc) {
864 if ($acc->is_retention == false && isset($acc->assets)) {
865 $acc->assets = $acc->assets - $retAmount;
866 $acc->amount = $acc->amount - $retAmount;
867 }
868 }

871 'recordsAccounting' => $accounts,
872 'accountingAccounts' => json_decode($this->getGroupAccountingAccount()),
