237// Returns a function, that, as long as it continues to be invoked, will not
238// be triggered. The function will be called after it stops being called for
239// N milliseconds. If `immediate` is passed, trigger the function on the
240// leading edge, instead of the trailing.
241
242function debounce(func, wait, immediate) {
243 var timeout;
244 return function() {
245 var context = this,
246 args = arguments;
247 clearTimeout(timeout);
248 timeout = setTimeout(function() {
249 timeout = null;
250 if (!immediate) func.apply(context, args);
251 }, wait);
252 if (immediate && !timeout) func.apply(context, args);
253 };
254};
