407 function uploadProfileImage() {
408 $('.preloader').show();
409 var url = $("#formImgProfile").attr('action');
410 var formData = new FormData();
411 var imageFile = document.querySelector('#profile_image');
412 if (imageFile.files[0].type.indexOf("image") < 0) {
413 $.gritter.add({
414 title: '{{ __('Alerta') }}!',
415 text: "{{ __('El formato de la imagen es incorrecto') }}",
416 class_name: 'growl-warning',
417 image: "{{ asset('images/screen-warning.png') }}",
418 sticky: false,
419 time: 2500
420 });
421 $('.preloader').fadeOut(2000);
422 return false;
423 }
424 formData.append("image", imageFile.files[0]);
425 axios.post(url, formData, {
426 headers: {
427 'Content-Type': 'multipart/form-data'
428 }
429 }).then(response => {
430 var up = response.data;
431 if (up.result) {
432 $.gritter.add({
433 title: '{{ __('Exito') }}!',
434 text: "{{ __('La imagen de perfil ha sido actualizada') }}",
435 class_name: 'growl-success',
436 image: "{{ asset('images/screen-ok.png') }}",
437 sticky: false,
438 time: 2500
439 });
440 axios.get('/get-image/' + up.image_id).then(response => {
441 var image = response.data.image;
442 $(".img-profile").attr('src', `${window.app_url}/${image.url}`);
443 $(".img-profile-mini").attr('src', `${window.app_url}/${image.url}`);
444 axios.post('{{ route('profiles.store') }}', {
445 first_name: $("#first_name").val(),
446 user_id: {{ auth()->user()->id }},
447 image_id: image.id
448 }).then(response => {
449 $('.preloader').fadeOut(2000);
450 }).catch(error => {
451 logs('profile', 781, error);
452 $('.preloader').fadeOut(2000);
453 });
454 }).catch(error => {
455 logs('profile', 784, error);
456 $('.preloader').fadeOut(2000);
457 });
458 }
459 $('.preloader').fadeOut(2000);
460 }).catch(error => {
461 logs('profile', 788, error);
462 $('.preloader').fadeOut(2000);
463 });
464 }
