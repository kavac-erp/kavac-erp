220  public function getTranslateFormulaAttribute()
221  {
222  $parameters = new PayrollAssociatedParametersRepository();
223  $formula = str_replace('if', 'Si', $this->formula);
224  $typesParameters = ['associatedBenefit', 'associatedVacation', 'associatedWorkerFile', 'parameter', 'concept', 'tabulator', 'ari_register'];
225 
226  foreach ($typesParameters as $typeParameter) {
227  if (in_array($typeParameter, ['parameter', 'concept', 'tabulator', 'ari_register'])) {
228  if ($typeParameter == 'parameter') {
229  $types = Parameter::where(
230  [
231  'required_by' => 'payroll',
232  'active' => true,
233  ]
234  )->where('p_key', 'like', 'global_parameter_%')->get();
235  foreach ($types as $type) {
236  $jsonValue = json_decode($type->p_value);
237  $formula = str_replace('parameter(' . $jsonValue->id . ')', $jsonValue->name, $formula);
238  }
239  } elseif ($typeParameter == 'concept') {
240  $types = PayrollConcept::all();
241  foreach ($types as $type) {
242  $formula = str_replace('concept(' . $type['id'] . ')', $type['name'], $formula);
243  }
244  } elseif ($typeParameter == 'tabulator') {
245  $types = PayrollSalaryTabulator::all();
246  foreach ($types as $type) {
247  $formula = str_replace('tabulator(' . $type['id'] . ')', $type['name'], $formula);
248  }
249  } elseif ($typeParameter == 'ari_register') {
250  $formula = str_replace('ari_register', 'Registro ARI', $formula);
251  }
252  } else {
253  $types = $parameters->loadData($typeParameter);
254  foreach ($types as $type) {
255  if (empty($type['children'])) {
256  $formula = str_replace($type['id'], $type['name'], $formula);
257  } else {
258  foreach ($type['children'] as $children) {
259  $formula = str_replace($children['id'], $children['name'], $formula);
260  }
261  }
262  }
263  }
264  }
265  return $formula;
