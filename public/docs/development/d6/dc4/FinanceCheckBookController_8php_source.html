72 'finance_bank_id' => $bank->id, 'finance_bank_account_id' => $account->id,

99 * @param Request $request Datos de la petición

103 public function store(Request $request)
104 {
105 $this->validate($request, [
106 'code' => ['required'],
107 'finance_bank_account_id' => ['required'],
108 'numbers' => ['required', 'array', 'min:1']
109 ], [
110 'code.required' => ('El campo serial / código es obligatorio.'),
111 'finance_bank_account_id.required' => ('El campo banco es obligatorio.'),
112 'numbers.required' => ('El campo número de cheque es obligatorio. Se deben registrar los números de cheques, pulsando el icono +'),
113 'numbers.min' => ('Debe ingresar al menos 1 numero de cheque.'),
114 ]);
115
116 foreach ($request->numbers as $number) {
117 if ($number == "") {
118 $error[0] = "El campo número de cheque es obligatorio. Se deben registrar los números de cheques, pulsando el icono +";
119 return response()->json(['result' => true, 'errors' => ["code" => $error]], 422);
120 }
121
122 //consulta que no exista el campo numero de cheque repetido con el codigo en la base de datos.
123 $checksnumber = FinanceCheckBook::where('number', $number)->where('code', $request->code)->first();
124 if ($checksnumber) {
125 $error[0] = "El campo Serial / Código o el Cheque # ya ha sido registrado en el sistema";
126 return response()->json(['result' => true, 'errors' => ["code" => $error]], 422);
127 }
128
129 //consulta que no exista el campo numero de cheque esta repetido en el formulario
130 if (count($request->numbers) > count(array_unique($request->numbers))) {
131 $error[0] = "El campo numero de cheque esta repetido en el formulario";
132 return response()->json(['result' => true, 'errors' => ["code" => $error]], 422);
133 }
134 }
135
136 //consulta que no exista el campo codigo repetido en base de datos.
137 $checkscode = FinanceCheckBook::where('code', $request->code)->first();
138 if ($checkscode) {
139 $error[0] = "El campo código chequera ya ha sido registrado";
140 return response()->json(['result' => true, 'errors' => ["code" => $error]], 422);
141 }
142
143 foreach ($request->numbers as $number) {
144 FinanceCheckBook::create([
145 'code' => $request->code,
146 'finance_bank_account_id' => $request->finance_bank_account_id,
147 'number' => $number,
148 ]);
149 }
150
151
152 return response()->json(['result' => true, 'message' => 'Success'], 200);
153 }
