109 /**
110 * Guarda información de las configuraciones de presupuesto
111 *
112 * @author Ing. Roldan Vargas <rvargas@cenditel.gob.ve> | <roldandvg@gmail.com>
113 *
114 * @param Request $request Datos de la petición
115 *
116 * @return Renderable
117 */
118 public function store(Request $request)
119 {
120 /* Reglas de validación para la configuración de códigos */
121 $request->validate([
122 'formulations_code' => [new CodeSettingRule()],
123 'compromises_code' => [new CodeSettingRule()],
124 'caused_code' => [new CodeSettingRule()],
125 'payed_code' => [new CodeSettingRule()],
126 'transfers_code' => [new CodeSettingRule()],
127 'reductions_code' => [new CodeSettingRule()],
128 'credits_code' => [new CodeSettingRule()],
129 'budgetary_availabilities_code' => [new CodeSettingRule()],
130 ]);
131
132 /* Arreglo con información de los campos de códigos configurados */
133 $codes = $request->input();
134 /* Define el estatus verdadero para indicar que no se ha registrado información */
135 $saved = false;
136
137 foreach ($codes as $key => $value) {
138 /* Define el modelo al cual hace referencia el código */
139 $model = '';
140
141 if ($key !== '_token' && !is_null($value)) {
142 list($table, $field) = explode("_", $key);
143 list($prefix, $digits, $sufix) = CodeSetting::divideCode($value);
144
145 if ($table === "formulations") {
146 /* Define el modelo asociado a la formulación de presupuesto */
147 $model = BudgetSubSpecificFormulation::class;
148 } elseif (in_array($key, ['transfers_code', 'reductions_code', 'credits_code'])) {
149 /* Define el modelo asociado a las modificaciones presupuestarias */
150 $model = BudgetModification::class;
151 /* Define la tabla asociada a las modificaciones presupuestarias */
152 $table = 'modifications';
153 if ($key === 'transfers_code') {
154 /* Define el tipo de registro como transferencia entre presupuestos */
155 $type = 'budget.transfers';
156 } elseif ($key === 'reductions_code') {
157 /* Define el tipo de registro como reducciones presupuestarias */
158 $type = 'budget.reductions';
159 } elseif ($key === 'credits_code') {
160 /* Define el tipo de registro como créditos adicionales */
161 $type = 'budget.aditional-credits';
162 }
163 } elseif ($table === "compromises") {
164 $model = BudgetCompromise::class;
165 } elseif ($key === 'budgetary_availabilities_code') {
166 /* Define el modelo asociado a las disponibilidades presupuestarias */
167 $model = \Modules\Purchase\Models\PurchaseBudgetaryAvailability::class;
168 }
169
170 // Si el caso es budgetary_availabilities_code.
171 if ($key === 'budgetary_availabilities_code') {
172 // Buscar en la base de datos la existencia de un código igual.
173 $codeSetting = CodeSetting::where([
174 'module' => 'purchase',
175 'table' => 'purchase_budgetary_availabilities',
176 'field' => 'code',
177 'type' => null
178 ])->first();
179
180 /* Si no se encuentra ninguna configuración de código que
181 coincida se crea un registro utilizando el método create()
182 del modelo CodeSetting.
183 */
184 if (!isset($codeSetting)) {
185 $codeSetting = CodeSetting::create([
186 'module' => 'purchase',
187 'table' => 'purchase_budgetary_availabilities',
188 'field' => 'code',
189 'type' => null,
190 'format_prefix' => $prefix,
191 'format_digits' => $digits,
192 'format_year' => $sufix,
193 'model' => $model
194 ]);
195 }
196 } else {
197 // Otros casos que no sean de budgetary_availabilities_code.
198 $codeSetting = CodeSetting::where([
199 'module' => 'budget',
200 'table' => 'budget_' . $table,
201 'field' => $field,
202 'type' => (isset($type)) ? $type : null
203 ])->first();
204
205 if (!isset($codeSetting)) {
206 $codeSetting = CodeSetting::create([
207 'module' => 'budget',
208 'table' => 'budget_' . $table,
209 'field' => $field,
210 'type' => (isset($type)) ? $type : null,
211 'format_prefix' => $prefix,
212 'format_digits' => $digits,
213 'format_year' => $sufix,
214 'model' => $model
215 ]);
216 }
217 }
218
219 /* Define el estatus verdadero para indicar que se ha registrado información */
220 $saved = true;
221 }
222 }
223
224 if ($saved) {
225 $request->session()->flash('message', ['type' => 'store']);
226 }
227
228 return redirect()->back();
229 }
