219  return false;
220  } else if (value === 'C') {
221  vm.$parent.record[vm.formulaInput] = '';
222  return false;
223  }
224 
225  if (formulaDisplay.length === 0 && symbols.includes(value)) {
226  vm.showMessage(
227  'custom', 'Fórmula Inválida', 'warning', 'screen-warning',
228  'No esta permitido indicar símbolos como primer elemento de la fórmula'
229  );
230  return false;
231  } else if (symbols.includes(formulaDisplay.slice(-1)) && symbols.includes(value)) {
232  vm.showMessage(
233  'custom', 'Fórmula Inválida', 'warning', 'screen-warning',
234  'No esta permitido indicar símbolos de forma consecutiva'
235  );
236  return false;
237  }
238 
239  if (value === 0 && vm.$parent.record[vm.formulaInput].slice(-1) === '/') {
240  vm.showMessage(

246  }
