172  $flag1 = true;
173  } elseif ($payExec->status == 'AN') {
174  $flag3 = true;
175  } elseif ($payExec->status == 'PE') {
176  $flag2 = true;
177  }
178  }
179  }
180  if ($flag1) {
181  return 'PA';
182  }
183  if ($flag2) {
184  return 'PE';
185  }
186  if ($flag3) {
187  return 'AN';
188  }
189  return ""; //No hay emisiones de pagos relacionadas.
