187 public function getStatusPaymentExecuteAttribute()

192 $paymentExecute = $this->financePaymentExecute()->get();
193 if (isset($paymentExecute)) {
194 foreach ($paymentExecute as $payExec) {
195 if ($payExec->status == 'PP' || $payExec->status == 'PA') {
196 $flag1 = true;
197 } elseif ($payExec->status == 'AN') {
198 $flag3 = true;
199 } elseif ($payExec->status == 'PE') {
200 $flag2 = true;
201 }
202 }
203 }
204 if ($flag1) {
205 return 'PA';
206 }
207 if ($flag2) {
208 return 'PE';
209 }
210 if ($flag3) {
211 return 'AN';
212 }
213 return ""; //No hay emisiones de pagos relacionadas.
