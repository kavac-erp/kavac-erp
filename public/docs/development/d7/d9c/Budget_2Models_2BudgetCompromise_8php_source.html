141 /**
142 * Método que permite obtener el estatus de un compromiso
143 *
144 * @author Daniel Contreras <dcontreras@cenditel.gob.ve>
145 *
146 * @return string Retorna el estatus de un compromiso
147 */
148 public function getStatusAttribute()
149 {
150 $stages = $this->budgetStages()->orderBy('type', 'asc')->get();
151 $status_action = $this->documentStatus->action;
152
153 if ($status_action == 'AN') {
154 // Anulado
155 return 'AN';
156 }
157
158 $status = $status_action;
159 $stagesPagAmount = 0;
160 $stagesComAmount = 0;
161
162 if (Module::has('Finance') && Module::isEnabled('Finance')) {
163 foreach ($stages as $stage) {
164 if ($stage->type == 'CAU') {
165 $status = $stage->type ;
166 } elseif ($stage->type == 'PAG') {
167 $stagesPagAmount += $stage->amount;
168 } elseif ($stage->type == 'COM') {
169 $stagesComAmount += $stage->amount;
170 }
171 }
172
173 if ($stagesComAmount > 0 && $stagesPagAmount > 0 && (string)$stagesPagAmount == (string)$stagesComAmount) {
174 $status = 'PA';
175 } elseif ($status_action == 'PR' || $status_action == 'EL') {
176 // En Proceso o Elaborado El estatus queda pendiente
177 $status = 'PE';
178 }
179 }
180
181 return $status;
182 }

194 $source = Source::query()
195 ->with('receiver.associateable')
196 ->where('sourceable_id', $this->id)
197 ->where('sourceable_type', BudgetCompromise::class)
198 ->orderBy('id', 'desc')
199 ->first();
200
201 $receiver = $source ? $source->receiver : null;
202
203 if ($receiver == null) {
204 if (Module::has('Purchase') && Module::isEnabled('Purchase')) {
205 $purchaseReceiver = \Modules\Purchase\Models\PurchaseDirectHire::with('purchaseSupplier')
206 ->where('code', $this->document_number)
207 ->first();
208 if ($purchaseReceiver != null) {
209 $supplierId = $purchaseReceiver->purchaseSupplier->id;
210
211 $receiver = Receiver::with(['receiverable', 'associateable'])
212 ->where('receiverable_id', $supplierId)
213 ->where('receiverable_type', 'Modules\Purchase\Models\PurchaseSupplier')
214 ->first();
