104 $request->end_date

109 "created_at",

112 now()

114 );
115 }
116 }
117 }
118 if ($request->type_search == "mes") {
119 if (!is_null($request->mes_id)) {
120 if (!is_null($request->year)) {
121 $fields = $fields->whereMonth(
122 'created_at',

169

175 $currentFiscalYear->year : date('Y')),
176 SaleWarehouseReport::class,

178 );
179
180 $report = SaleWarehouseReport::create([
181 'code' => $code,
182 'type_report' => $request->current,
183 'institution_id' => $institution->id,
184 'filename' => $filename
185 ]);
186
187 $pdf->setConfig([
188 'institution' => Institution::first(),
189 'filename' => $filename
190 ]);
191 $pdf->setHeader('Reporte de inventario de productos');
192 $pdf->setFooter();
193 $pdf->setBody('sale::pdf.inventory-products', true, [
194 'pdf' => $pdf,
195 'inventory_products' => $request->sale_warehouse_products
196 ]);
197
198 $url = '/sale/reports/show/' . $report->code;
199 return response()->json(['result' => true, 'redirect' => $url], 200);
