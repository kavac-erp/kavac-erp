95 <script>
96 /**

98 *
99 * @author Ing. Roldan Vargas <rvargas@cenditel.gob.ve> | <roldandvg@gmail.com>

103 function delete_backup(filename) {

106 location.href="/backup/delete/" + filename;
107 }
108 });
109 }
110
111 /**
112 * Ejecuta la acción corrspondiente para la restauración de registros
113 *
114 * @author Ing. Roldan Vargas <rvargas@cenditel.gob.ve> | <roldandvg@gmail.com>
115 *
116 * @param {string} filename Nombre del archivo con el respaldo a restaurar
117 */
118 function restore_backup(filename) {
119 bootbox.confirm({
120 title: '{{ __('Restaurar datos') }}',
121 message: `<p class='text-justify'>
122 {{ __('¿Está seguro de restaurar este respaldo? ') }}
123 {{ __('Se eliminará cualquier dato existente y se restaurará la información con este respaldo. ') }}
124 {{ __('Deberá autenticarse nuevamente luego de la restauración de la base de datos.') }}
125 </p>
126 <p class='text-justify'>
127 <b class='text-red'>¡ATENCIÓN!</b>
128 Este proceso puede ocasionar perdida parcial o total de los datos, proceda si esta
129 totalmente seguro.
130 </p>`,
131 buttons: {
132 confirm: {
133 label: '<i class="fa fa-upload"></i> Restaurar',
134 className: 'btn btn-lg btn-primary'
135 },
136 cancel: {
137 label: '<i class="fa fa-ban"></i> Cancelar',
138 className: 'btn btn-lg'
139 }
140 },
141 callback: (result) => {
142 if (result) {
143 /** Muestra el mensaje de espera */
144 $('.preloader').show();
145 axios.post(
146 '/backup/restore', {filename: filename}
147 ).then(response => {
148 if (response.data.result) {
149 showMessageRestore(true);
150 return true;
151 }
152 /** Oculta el mensaje de espera */
153 $('.preloader').fadeOut(2000);
154 let msg = 'No fue posible realizar la restauración del respaldo, revise los logs del sistema';
155 showMessageRestore(false, msg);
156 }).catch(error => {
157 /** Oculta el mensaje de espera */
158 $('.preloader').fadeOut(2000);
159 let errorObj = error.response;
160 if (errorObj.status === 500 && errorObj.data.message.includes("Undefined table")) {
161 showMessageRestore(false, errorObj.data.message);
162 }
163 });
164 }
165 }
166 });
167 }
168
169 /**
170 * [showMessageRestore description]
171 *
172 * @author Ing. Roldan Vargas <rvargas@cenditel.gob.ve> | <roldandvg@gmail.com>
173 *
174 * @param {boolean} success Determiuna si la restauración fue exitosa
175 * @param {String} message Mensaje a ser mostrado
176 */
177 function showMessageRestore(success, message = '') {
178 $.gritter.add({
179 title: (success) ? '{{ __('Éxito') }}' : '{{ __('Alerta!') }}',
180 text: (success) ? "{{ __('Se ha restaurado la base de datos satisfactoriamente') }}" : message,
181 class_name: (success) ? 'growl-success' : 'growl-danger',
182 image: (success) ? "{{ asset('images/screen-ok.png') }}" : "{{ asset('images/screen-error.png') }}",
183 sticky: false,
184 time: 2500
185 });
186 }
187 </script>
