161 {
162 if ($item !== '00') {
163 $parent = self::where('group', $group);
164 if ($generic !== '00') {
165 $parent = $parent->where('item', $item);
166 if ($specific !== '00') {
167 $parent = $parent->where('generic', $generic);
168 if ($subspecific !== '00') {
169 $parent = $parent->where('specific', $specific);
170 } else {
171 $parent = $parent->where('subspecific', '00');
172 }
173 } else {
174 $parent = $parent->where('specific', '00');
175 }
176 } else {
177 $parent = $parent->where('generic', '00');
178 }
179 }
180
181 if (!isset($parent)) {
182 return false;
183 }
184
185 return $parent->first();
186 }
