324 /**
325 * Valida un rango de fechas
326 *
327 * @param \Illuminate\Http\Request $request Datos e la petición
328 * @param string $attribute Nombre del atributo
329 * @param integer $value Identificador del registro
330 * @param mixed $fail Registros fallidos
331 * @param mixed $querySet Objeto con los datos de la consulta
332 *
333 * @return void
334 */
335 private function validateDateInRange(Request $request, $attribute, $value, $fail, $querySet)
336 {
337 $currentIndex = explode('.', $attribute)[1];
338 if (isset($value)) {
339 $entity = $querySet->find($value);
340 $startDateEntity = Carbon::parse($entity->start_date);
341 $endDateEntity = Carbon::parse($entity->end_date);
342 $startDateTask = Carbon::parse($request->tasks[$currentIndex]['start_date']);
343 $endDateTask = Carbon::parse($request->tasks[$currentIndex]['end_date']);
344 if (!$startDateTask->between($startDateEntity, $endDateEntity) && !$endDateTask->between($startDateEntity, $endDateEntity)) {
345 $fail("La fecha de inicio y fin de la tarea " . $currentIndex + 1 . " debe estar entre {$startDateEntity->format('d/m/Y')} y {$endDateEntity->format('d/m/Y')}.");
346 }
347 }
