129 public function filterRecords(Request $request)
130 {
131 $filter = $request->all();
132
133 $records = SaleService::with(['SaleServiceRequirement','saleClient', 'payrollStaff']);
134 if ($filter['filterDate'] == 'specific') {
135 if ($filter['dateIni'] != null && $filter['dateEnd'] != null) {
136 $records->whereDate('created_at', '>=', $filter['dateIni'])->whereDate('created_at', '<=', $filter['dateEnd']);
137 }
138 } elseif ($filter['filterDate'] == 'general') {
139 if ($filter['year_init'] != null && $filter['year_end'] != null && $filter['month_init'] != null && $filter['month_end'] != null) {
140 $records->whereYear('created_at', '>=', $filter['year_init'])->whereYear('created_at', '<=', $filter['year_end'])
141 ->whereMonth('created_at', '>=', $filter['month_init'])->whereMonth('created_at', '<=', $filter['month_end']);
142 }
143 }
144
145 if ($filter['status'] != null && $filter['status'] != 'Todos') {
146 $records->where('status', $filter['status']);
147 }
148 return response()->json([
149 'records' => $records->get(),
150 'message' => 'success'], 200);
151 }
