110 ]);
111
112 $products = WarehouseInventoryProduct::where('id', $request->warehouse_products[$i]['id'])->with('warehouseProduct')->first();
113
114 $messages = array_merge($messages, [
115 'warehouse_products.' . $i . '.requested.max' => 'El producto "' . $products->warehouseProduct->name . '" no posee suficiente existencia en almacén'
116 ]);
117 }
118
119
120 $this->validate($request, $validateRules, $messages);
121
122 $codeSetting = CodeSetting::where('table', 'warehouse_requests')->first();
123 if (is_null($codeSetting)) {
124 $request->session()->flash('message', [
125 'type' => 'other', 'title' => 'Alerta', 'icon' => 'screen-error', 'class' => 'growl-danger',
126 'text' => 'Debe configurar previamente el formato para el código a generar'
127 ]);
128 return response()->json(['result' => false, 'redirect' => route('warehouse.setting.index')], 200);
129 }
130
131 $code = generate_registration_code(
132 $codeSetting->format_prefix,
133 strlen($codeSetting->format_digits),
134 (strlen($codeSetting->format_year) == 2) ? date('y') : date('Y'),
135 $codeSetting->model,
136 $codeSetting->field
137 );
138
139
140 DB::transaction(function () use ($request, $code) {
141 $data_request = WarehouseRequest::create([
142 'code' => $code,
143 'request_date' => $request->input('request_date'),
144 'state' => 'Pendiente',
145 'department_id' => $request->input('department_id'),
146 'motive' => $request->input('motive'),
147 'payroll_staff_id' => $request->input('payroll_staff_id')
148 ]);
149
150 foreach ($request->warehouse_products as $product) {
151 $inventory_product = WarehouseInventoryProduct::find($product['id']);
152 if (!is_null($inventory_product)) {
153 $exist_real = $inventory_product->exist - $inventory_product->reserved;
154 if ($exist_real >= $product['requested']) {
155 WarehouseInventoryProductRequest::create([
156 'warehouse_inventory_product_id' => $inventory_product->id,
157 'warehouse_request_id' => $data_request->id,
158 'quantity' => $product['requested'],
159 'new_exist' => $exist_real - $product['requested'],
160 ]);

163 DB::rollback();

165 } else {
166 /* Si no existe el registro se revierten los cambios */
167 DB::rollback();
168 }
169 }
170 });
171 $warehouse_request = WarehouseRequest::where('code', $code)->first();
172 if (is_null($warehouse_request)) {
173 $request->session()->flash(
174 'message',
175 [
176 'type' => 'other',
177 'title' => 'Alerta',
178 'icon' => 'screen-error',

182 );
