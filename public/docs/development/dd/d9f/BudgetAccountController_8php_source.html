141 public function store(Request $request)
142 {
143 $this->validate($request, $this->validate_rules, $this->messages);
144 list($group, $item, $generic, $specific, $subspecific) = explode(".", $request->code);
145
146 /* Objeto que contiene los datos de la cuenta ya registrada si existe */
147 $budgetAccount = BudgetAccount::where('group', $group)
148 ->where('item', $item)
149 ->where('generic', $generic)
150 ->where('specific', $specific)
151 ->where('subspecific', $subspecific)
152 ->where('active', true)->first();
153
154 /*
155 * Si la cuenta a registrar ya existe en la base de datos y la nueva cuenta se indica como activa,
156 * se desactiva la cuenta anterior
157 */
158 if ($budgetAccount && $request->active !== null) {
159 /* define si la cuenta esta activa */
160 $budgetAccount->active = false;
161 $budgetAccount->save();
162 }
163
164 /* Objeto con información de la cuenta de nivel superior, si existe */
165 $parent = BudgetAccount::getParent($group, $item, $generic, $specific, $subspecific);
166
167 /* Registra la nueva cuenta presupuestaria */
168 BudgetAccount::create([
169 'group' => $group,
170 'item' => $item,
171 'generic' => $generic,
172 'specific' => $specific,
173 'subspecific' => $subspecific,
174 'denomination' => $request->denomination,
175 'resource' => ($request->account_type == "resource"),
176 'egress' => ($request->account_type == "egress"),
177 'active' => ($request->active !== null),
178 'original' => ($request->original !== null),
179 'parent_id' => ($parent == false) ? null : $parent->id,
180 'disaggregate_tax' => ($request->disaggregate_tax !== null) ? true : false
181 ]);
182
183 $request->session()->flash('message', ['type' => 'store']);
184 return redirect()->route('budget.accounts.index');
185 }

382 public function setChildrenAccount($parent_id)
383 {
384 /* Objeto con información de la cuenta presupuestaria de nivel superior */
385 $parent = BudgetAccount::find($parent_id);
386
387 /* Contiene el código del ítem */
388 $item = $parent->item;

394 $subspecific = $parent->subspecific;
395
396 if ($parent->item === "00") {
397 /* Contiene información de la cuenta presupuestaria por el código del ítem */
398 $currentItem = BudgetAccount::where(['group' => $parent->group])->orderBy('item', 'desc')->first();
399 /* Contiene el código inmediatamente disponible para su registro */
400 $item = (strlen(intval($currentItem->item)) < 2 || intval($currentItem->item) < 99)
401 ? (intval($currentItem->item) + 1) : $currentItem->item;
402 /* Determina la longitud de la cadena para agregar un cero a la izquierda en
403 caso de requerirlo */
404 $item = (strlen($item) === 1) ? "0$item" : $item;
405 } elseif ($parent->generic === "00") {
406 /* Contiene información de la cuenta presupuestaria por el código de la genérica */
407 $currentGeneric = BudgetAccount::where(['group' => $parent->group, 'item' => $parent->item])

409 /* Contiene el código inmediatamente disponible para su registro */
410 $generic = (strlen(intval($currentGeneric->generic)) < 2 || intval($currentGeneric->generic) < 99)
411 ? (intval($currentGeneric->generic) + 1) : $currentGeneric->generic;
412 /* Determina la longitud de la cadena para agregar un cero a la izquierda en caso
413 de requerirlo */
414 $generic = (strlen($generic) === 1) ? "0$generic" : $generic;
415 } elseif ($parent->specific === "00") {
416 /* Contiene información de la cuenta presupuestaria por el código de la específica */
417 $currentSpecific = BudgetAccount::where([
418 'group' => $parent->group, 'item' => $parent->item, 'generic' => $parent->generic
419 ])->orderBy('specific', 'desc')->first();
