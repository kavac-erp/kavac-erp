376 /* Initialization method
377 ================================================== */
378 Plugin.prototype.init = function() {
379 var self = this;
380 if (this.options.urlCurl !== '') {
381 urlJson.googlePlus = this.options.urlCurl + '?url={url}&type=googlePlus'; // PHP script for GooglePlus...
382 urlJson.stumbleupon = this.options.urlCurl + '?url={url}&type=stumbleupon'; // PHP script for Stumbleupon...
383 }
384 $(this.element).addClass(this.options.className); //add class
385
386 //HTML5 Custom data
387 if (typeof $(this.element).data('title') !== 'undefined') {
388 this.options.title = $(this.element).attr('data-title');
389 }
390 if (typeof $(this.element).data('url') !== 'undefined') {
391 this.options.url = $(this.element).data('url');
392 }
393 if (typeof $(this.element).data('text') !== 'undefined') {
394 this.options.text = $(this.element).data('text');
395 }
396
397 //how many social website have been selected
398 $.each(this.options.share, function(name, val) {
399 if (val === true) {
400 self.options.shareTotal++;
401 }
402 });
403
404 if (self.options.enableCounter === true) { //if for some reason you don't need counter
405 //get count of social share that have been selected
406 $.each(this.options.share, function(name, val) {
407 if (val === true) {
408 //self.getSocialJson(name);
409 try {
410 self.getSocialJson(name);
411 } catch (e) {}
412 }
413 });
414 } else if (self.options.template !== '') { //for personalized button (with template)
415 this.options.render(this, this.options);
416 } else { // if you want to use official button like example 3 or 5
417 this.loadButtons();
418 }
419
420 //add hover event
421 $(this.element).hover(function() {
422 //load social button if enable and 1 time
423 if ($(this).find('.buttons').length === 0 && self.options.enableHover === true) {
424 self.loadButtons();
425 }
426 self.options.hover(self, self.options);
427 }, function() {
428 self.options.hide(self, self.options);
429 });
430
431 //click event

435 });

453 /* getSocialJson methode
454 ================================================== */
455 Plugin.prototype.getSocialJson = function(name) {
456 var self = this,
457 count = 0,
458 url = urlJson[name].replace('{url}', encodeURIComponent(this.options.url));
459 if (this.options.buttons[name].urlCount === true && this.options.buttons[name].url !== '') {
460 url = urlJson[name].replace('{url}', this.options.buttons[name].url);
461 }
462 //console.log('name : ' + name + ' - url : '+url); //debug
463 if (url != '' && self.options.urlCurl !== '') { //urlCurl = '' if you don't want to used PHP script but used social button
464 $.getJSON(url, function(json) {
465 if (typeof json.count !== "undefined") { //GooglePlus, Stumbleupon, Twitter, Pinterest and Digg
466 var temp = json.count + '';
467 temp = temp.replace('\u00c2\u00a0', ''); //remove google plus special chars
468 count += parseInt(temp, 10);
469 }
470 //get the FB total count (shares, likes and more)
471 else if (json.data && json.data.length > 0 && typeof json.data[0].total_count !== "undefined") { //Facebook total count
472 count += parseInt(json.data[0].total_count, 10);
473 } else if (typeof json[0] !== "undefined") { //Delicious
474 count += parseInt(json[0].total_posts, 10);
475 } else if (typeof json[0] !== "undefined") { //Stumbleupon
476 }
477 self.options.count[name] = count;
478 self.options.total += count;
479 self.renderer();
480 self.rendererPerso();
481 //console.log(json); //debug
482 })
483 .error(function() {
484 self.options.count[name] = 0;
485 self.rendererPerso();
486 });
487 } else {
488 self.renderer();
489 self.options.count[name] = 0;
490 self.rendererPerso();
491 }
492 };

494 /* launch render methode
495 ================================================== */
496 Plugin.prototype.rendererPerso = function() {
497 //check if this is the last social website to launch render
498 var shareCount = 0;
499 for (e in this.options.count) {
500 shareCount++;
501 }
502 if (shareCount === this.options.shareTotal) {
503 this.options.render(this, this.options);
504 }
505 };

507 /* render methode
508 ================================================== */
509 Plugin.prototype.renderer = function() {
510 var total = this.options.total,
511 template = this.options.template;
512 if (this.options.shorterTotal === true) { //format number like 1.2k or 5M
513 total = this.shorterTotal(total);
514 }
515
516 if (template !== '') { //if there is a template
517 template = template.replace('{total}', total);
518 $(this.element).html(template);
519 } else { //template by defaults
520 $(this.element).html(
521 '<div class="box"><a class="count" href="#">' + total + '</a>' +
522 (this.options.title !== '' ? '<a class="share" href="#">' + this.options.title + '</a>' : '') +
523 '</div>'
524 );
525 }
526 };

528 /* format total numbers like 1.2k or 5M
529 ================================================== */
530 Plugin.prototype.shorterTotal = function(num) {
531 if (num >= 1e6) {
532 num = (num / 1e6).toFixed(2) + "M"
533 } else if (num >= 1e3) {
534 num = (num / 1e3).toFixed(1) + "k"
535 }
536 return num;
537 };
