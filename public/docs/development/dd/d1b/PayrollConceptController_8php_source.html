566 /**
567 * Método que valida que la formula a agregar en el concepto sea una formula con un formato válido
568 *

570 *
571 * @return boolean Define si es una formula válida
572 */
573 public function validateFormula($formula)
574 {
575 $parameters = new PayrollAssociatedParametersRepository();
576 $typesParameters = [
577 'associatedBenefit',
578 'associatedVacation',
579 'associatedWorkerFile',
580 'parameter',
581 'concept',
582 'tabulator',
583 'ari_register'
584 ];
585
586 foreach ($typesParameters as $typeParameter) {
587 if (in_array($typeParameter, ['parameter', 'concept', 'tabulator', 'ari_register'])) {
588 if ($typeParameter == 'parameter') {
589 $types = Parameter::where(
590 [
591 'required_by' => 'payroll',
592 'active' => true,
593 ]
594 )->where('p_key', 'like', 'global_parameter_%')->get();
595 foreach ($types as $type) {
596 $jsonValue = json_decode($type->p_value);
597 $formula = str_replace('parameter(' . $jsonValue->id . ')', '1', $formula);
598 }
599 } elseif ($typeParameter == 'concept') {
600 $types = PayrollConcept::all();
601 foreach ($types as $type) {
602 $formula = str_replace('concept(' . $type['id'] . ')', '1', $formula);
603 }
604 } elseif ($typeParameter == 'tabulator') {
605 $types = PayrollSalaryTabulator::all();
606 foreach ($types as $type) {
607 $formula = str_replace('tabulator(' . $type['id'] . ')', '1', $formula);
608 }
609 } elseif ($typeParameter == 'ari_register') {
610 $formula = str_replace('ari_register', '1', $formula);
611 }
612 } else {
613 $types = $parameters->loadData($typeParameter);
614 foreach ($types as $type) {
615 if (empty($type['children'])) {
616 $formula = str_replace($type['id'], '1', $formula);
617 } else {
618 foreach ($type['children'] as $children) {
619 $formula = str_replace($children['id'], '1', $formula);
620 }
621 }
622 }
623 }
624 }
